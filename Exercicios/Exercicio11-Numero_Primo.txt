Exercício 11 - Número Primo
Algoritmo "NumeroPrimo"

Var
N, c, contDiv: inteiro

Inicio
c <- 1
Escreva("Digite um número: ")
Leia(N)
Repita
    Se (N % c = 0) entao
        contDiv <- contDiv + 1
    FimSe
    c <- c + 1
Ate (c > N)

Se (contDiv > 2) entao
    Escreva("Esse número NÃO é primo!")
senao
    Escreva("Esse número é primo!")
FimSe

Fimalgoritmo
