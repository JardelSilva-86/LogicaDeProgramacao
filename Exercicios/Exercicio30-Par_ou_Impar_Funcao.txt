Exercício 2 - Par ou Ímpar
Algoritmo "ParOuImpar"

Var
n: inteiro
r: caractere

Funcao ParOuImpar(v: inteiro) :Caractere
Inicio
    Se (v % 2 = 0) entao
        Retorne "Par"
    senao
        Retorne "Ímpar"
    FimSe
FimFuncao

Inicio
Escreva("Digite um número: ")
Leia(n)
r := ParOuImpar(n)
Escreva("O número",n, " é um valor ",r, "!")
