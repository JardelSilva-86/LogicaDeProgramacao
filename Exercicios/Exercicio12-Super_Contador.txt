    Escreval("|      MENU      |")
    Escreval("==================")
    Escreval("| [1] De 1 a 10  |")
    Escreval("| [2] De 10 a 1  |")

        c <- 10
        Repita
            Escreva(c,"..")
            c <- c - 1
        Ate(c <= 1)
    
    Caso 3

    OutroCaso
        Escreval("Invalido...")
    FimEscolha
