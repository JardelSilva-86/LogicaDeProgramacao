"Exercício 2 - Conta Negativo"
Algoritmo "ContaNegativo"

Var
c, TotN, n: inteiro

Inicio
c <- 1
TotN <- 0

Repita
    Escreva("Digite um número: ")
    Leia(n)
    Se (n < 0) entao
        TotN <-  TotN + 1
    FimSe
    c <- c + 1
Ate(c > 3)

Escreva("O total de números negativos são: ", TotN)

Fimalgoritmo
