"Exercício 1 - Some os Números"
Algoritmo "SomeOsNumeros"

Var
cont, n, s: inteiro

Inicio
    cont <- 1
    s <- 0
    Enquanto (cont <= 3) faca
        Escreva (" Digite o ", cont, "o. valor: ")
        Leia (n)
        s <- s + n
        cont <- cont + 1
    FimEnquanto
    Escreval (" A soma de todos os valores é: ", s)

Fimalgoritmo
